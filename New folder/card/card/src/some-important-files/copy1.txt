const YourComponent: React.FC = () => {
  // ... (previous code)

  const handleApplyCompanyToAll = () => {
    const firstCard = cards[0]; // Assuming at least one card exists
    if (!firstCard) return;

    setCards(prevCards => {
      const updatedCards = prevCards.map(card => ({
        ...card,
        selectedCompanyCheckboxes: firstCard.selectedCompanyCheckboxes,
        companyInputValues: firstCard.companyInputValues,
      }));
      return updatedCards;
    });
  };

  return (
    <div>
      {cards.map(card => (
        <div key={card.id}>
          {/* ... (other card rendering) */}
        </div>
      ))}
      {/* ... (Add Card button) */}
      <div>
        <input
          type="checkbox"
          checked={cards.every(card => card.applyCompanyToAll)} // Check if all cards have it checked
          onChange={() => {
            const updatedCards = cards.map(card => ({
              ...card,
              applyCompanyToAll: !cards.every(c => c.applyCompanyToAll),
            }));
            setCards(updatedCards);
            if (updatedCards.every(c => c.applyCompanyToAll)) {
              handleApplyCompanyToAll();
            }
          }}
        />
        <label>Apply Company information to all cards</label>
      </div>
    </div>
  );
};

export default YourComponent;
